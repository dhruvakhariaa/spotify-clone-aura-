- [ ] **Step 3: Add `buildUniverseArc`.** Order = full Indian opening block → an interleave/crossfade zone that ramps the English ratio → English tail.

```ts
/** Order tracks as a Hindi/Indian → English gradient arc (not a hard cut). */
export function buildUniverseArc(tracks: UniverseTrack[]): UniverseTrack[] {
  const tagged = tracks.map((t) => ({ ...t, language: t.language ?? languageForTrack(t) }));
  const indian = tagged.filter((t) => t.language === "indian");
  const english = tagged.filter((t) => t.language === "english");
  if (!indian.length || !english.length) return tagged;

  const openCount = Math.round(indian.length * 0.55); // pure Hindi opening
  const opening = indian.slice(0, openCount);
  const indianMid = indian.slice(openCount);
  const out: UniverseTrack[] = [...opening];

  // Crossfade zone: walk remaining Indian + English, raising English share.
  const total = indianMid.length + english.length;
  let ei = 0;
  let ii = 0;
  for (let step = 0; step < total; step += 1) {
    const englishBias = total > 1 ? step / (total - 1) : 1; // 0 → 1
    const pickEnglish = ei < english.length && (ii >= indianMid.length || Math.random() < englishBias);
    if (pickEnglish) out.push(english[ei++]);
    else out.push(indianMid[ii++]);
  }
  return out;
}
```
